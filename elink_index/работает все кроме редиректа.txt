
limit_conn_zone $binary_remote_addr zone=two:1m;


server {
    server_tokens off;
    proxy_read_timeout 2;
    proxy_connect_timeout 2;
    proxy_send_timeout 2;
    server_name e-lnk.ru;

    location /media/ {
        alias /home/gera/e-lnk_backend/static/media/;
    }

    location /static/admin/ {
        alias /home/gera/e-lnk_backend/static/admin/;
    }

    location /panel_admin {

        proxy_pass http://127.0.0.1:8000/panel_admin/;
    }

    location / {
        root /home/gera/e-lnk_front/dist;
        index index.html;
        try_files $uri $uri/ @redir;
        proxy_set_header        Host $host;
        proxy_set_header        X-Real-IP $remote_addr;
        proxy_set_header        X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header        X-Forwarded-Proto $scheme;
    }

    location /api/v1 {
        proxy_pass http://127.0.0.1:8000;
    }

    location /dist {
        root /home/gera/e-lnk_front;
        expires max;
    }
    location /fonts {
        root /home/gera/e-lnk_front/dist;
        expires max;
    }
    location /img {
        root /home/gera/e-lnk_front/dist;
        expires max;
    }
    location /js {
        root /home/gera/e-lnk_front/dist;
        expires max;
    }
    location /scss {
        root /home/gera/e-lnk_front/dist;
        expires max;
    }

    location @redir {
        proxy_pass http://127.0.0.1:8000;
    }
    location = /404 {
        root /home/gera/e-lnk_front/dist;
        index 404.html;
        try_files $uri $uri $host;
    }

    listen 443 ssl; # managed by Certbot
    ssl_certificate /etc/letsencrypt/live/e-lnk.ru/fullchain.pem; # managed by Certbot
    ssl_certificate_key /etc/letsencrypt/live/e-lnk.ru/privkey.pem; # managed by Certbot
    include /etc/letsencrypt/options-ssl-nginx.conf; # managed by Certbot
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem; # managed by Certbot

}
server {
    if ($host = e-lnk.ru) {
        return 301 https://$host$request_uri;
    } # managed by Certbot


    server_name e-lnk.ru;
    listen 80;
    return 404; # managed by Certbot


}
